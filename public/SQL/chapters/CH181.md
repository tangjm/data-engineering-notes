CH181

Count app releases per month

```sql
WITH app_releases AS (
	SELECT
		app_version,
		MIN(created_at) AS release_date
	FROM
		adjust.callbacks
	GROUP BY
		1
	ORDER BY
		string_to_array(app_version, '.')
)
SELECT
	TO_CHAR(release_date, 'yyyy-mm') AS year_month,
	COUNT(app_version) AS releases
FROM
	app_releases
GROUP BY
	1
ORDER BY
	1
```
